-- CURRENT VS PREVIOUS SAI CHANGE

SELECT R1.RCRAPP1_PIDM

FROM RCRAPP1 R1

JOIN RORSTAT RS
	ON RS.RORSTAT_PIDM = R1.RCRAPP1_PIDM
		AND RS.RORSTAT_AIDY_CODE = R1.RCRAPP1_AIDY_CODE

WHERE R1.RCRAPP1_CREATE_DATE > RS.RORSTAT_PCKG_COMP_DATE
	AND R1.RCRAPP1_CURR_REC_IND = 'Y'
	AND R1.RCRAPP1_AIDY_CODE = :AIDY
	AND R1.RCRAPP1_PIDM = :PIDM
	AND NOT EXISTS (
		/* FIND A MATCH WHERE CURRENT SAI = PREVIOUS SAI FOR THIS EXACT CURRENT RECORD.

IF SUCH A MATCH EXISTS, WE EXCLUDE THE PIDM; OTHERWISE WE RETURN IT. */
		SELECT 1
		
		FROM (
			/* CURRENT SAI FOR THE EXACT CURRENT RCRAPP1 ROW */
			SELECT A.RCRAPP2_PELL_PGI AS CURRENT_SAI
			
			FROM RCRAPP2 A
			
			WHERE A.RCRAPP2_PIDM = R1.RCRAPP1_PIDM
				AND A.RCRAPP2_AIDY_CODE = R1.RCRAPP1_AIDY_CODE
				AND A.RCRAPP2_INFC_CODE = R1.RCRAPP1_INFC_CODE
				AND A.RCRAPP2_SEQ_NO = R1.RCRAPP1_SEQ_NO
			) CUR
		
		JOIN (
			/* PREVIOUS SAI ON THE MOST RECENT NON-CURRENT RCRAPP1 ROW */
			SELECT C.RCRAPP2_PELL_PGI AS PREVIOUS_SAI
			
			FROM RCRAPP2 C
			
			JOIN RCRAPP1 D
				ON D.RCRAPP1_PIDM = C.RCRAPP2_PIDM
					AND D.RCRAPP1_AIDY_CODE = C.RCRAPP2_AIDY_CODE
					AND D.RCRAPP1_INFC_CODE = C.RCRAPP2_INFC_CODE
					AND D.RCRAPP1_SEQ_NO = C.RCRAPP2_SEQ_NO
			
			WHERE D.RCRAPP1_PIDM = R1.RCRAPP1_PIDM
				AND D.RCRAPP1_AIDY_CODE = R1.RCRAPP1_AIDY_CODE
				AND D.RCRAPP1_INFC_CODE = R1.RCRAPP1_INFC_CODE
				AND D.RCRAPP1_CURR_REC_IND = 'N'
				AND D.RCRAPP1_SEQ_NO = (
					SELECT MAX(N.RCRAPP1_SEQ_NO)
					
					FROM RCRAPP1 N
					
					WHERE N.RCRAPP1_PIDM = R1.RCRAPP1_PIDM
						AND N.RCRAPP1_AIDY_CODE = R1.RCRAPP1_AIDY_CODE
						AND N.RCRAPP1_INFC_CODE = R1.RCRAPP1_INFC_CODE
						AND N.RCRAPP1_CURR_REC_IND = 'N'
					)
			) PREV
			ON (
					PREV.PREVIOUS_SAI = CUR.CURRENT_SAI
					OR (
						PREV.PREVIOUS_SAI IS NULL
						AND CUR.CURRENT_SAI IS NULL
						)
					)
		)
